- árvore binária cheia
- árvore binária completa

    A cheia é aquela em que todo nó possui exatamente 0 ou 2 filhos e a completa 
possui todos os níveis completos, exceto possivelmente o último.

08. Explique as diferenças entre os percursos:
- pré-ordem: Ela percorre a árvore começando pela raiz, depois vai para o lado da esquerda
e por fim para o lado da direita.

- em ordem: Percorre a árvore começando pelo lado da esquerda, depois vai para a raiz da árvore
e termina no lado da direita.

- pós-ordem: Começa a percorrer pelo lado da esquerda, depois vai para o lado direito e 
termina na raiz da árvore.

 inserções e remoções de dados de forma dinâmica.

IMPLEMENTAÇÃO DE ÁRVORES BINÁRIAS

36. Implemente a estrutura de um nó de árvore binária em C.

#include <stdio.h>
#include <stdlib.h>

typedef struct No {
    int valor;
    struct No *esq;
    struct No *dir;
} No;

No* criarNo(int valor) {
    No *novo = (No*) malloc(sizeof(No));

    if (novo != NULL) {
        novo->valor = valor;
        novo->esq = NULL;
        novo->dir = NULL;
    }

    return novo;
}

37. Crie uma função para inserir nós manualmente em uma árvore binária.

void inserirManual(No *pai, int valor, char lado) {
    if (pai == NULL) {
        printf("Nó pai inexistente!\n");
        return;
    }

    if (lado == 'E') {
        if (pai->esq == NULL)
            pai->esq = criarNo(valor);
        else
            printf("Filho esquerdo já existe!\n");
    }
    else if (lado == 'D') {
        if (pai->dir == NULL)
            pai->dir = criarNo(valor);
        else
            printf("Filho direito já existe!\n");
    }
    else {
        printf("Lado inválido! Use 'E' ou 'D'.\n");
    }
}

41. Crie uma função que conte o número de nós da árvore binária.

int contarNos(No *raiz) {
    if (raiz == NULL)
        return 0;

    return 1 + contarNos(raiz->esq) + contarNos(raiz->dir);
}

42. Implemente uma função que calcule a altura da árvore binária.

int altura(No *raiz) {
    if (raiz == NULL)
        return 0;

    int altEsq = altura(raiz->esq);
    int altDir = altura(raiz->dir);

    if (altEsq > altDir)
        return altEsq + 1;
    else
        return altDir + 1;
}

43. Implemente uma função que conte quantos nós folhas existem.

int contarFolhas(No *raiz) {
    if (raiz == NULL)
        return 0;

    if (raiz->esq == NULL && raiz->dir == NULL)
        return 1;

    return contarFolhas(raiz->esq) + contarFolhas(raiz->dir);
}

45. Implemente uma função que verifique se duas árvores binárias são iguais.

int saoIguais(No *a, No *b) {
    if (a == NULL && b == NULL)
        return 1;

    if (a == NULL || b == NULL)
        return 0;

    return (a->valor == b->valor) &&
           saoIguais(a->esq, b->esq) &&
           saoIguais(a->dir, b->dir);
